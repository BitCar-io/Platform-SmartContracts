pragma solidity ^0.4.24;
 
/**
 * @title Multisignature
 * @notice Contract that implements the basic functionality for multisignature 
 * function execution
 * TODO: Implement BlockNumber based expiration mechanism
 */
contract Multisignature {
 
    event MultisigExecutionRequest(address initiatedBy, uint256 transactionId);
    event MultisigExecutionApproval(address signedBy, uint256 transactionId);
    event MultisigExecutionCompleted(uint256 transactionId);
    event MultisigExecutionCancelled(uint256 transactionId);
 
    struct MultisigTransaction {
        address initiatedBy;
        uint8 signatureCount;
        uint8 minSignatureCount;
        mapping (address => uint8) signatures;
    }
    mapping(uint256 => MultisigTransaction) private multisigTransactions;
    uint256 private transactionIdx;
    uint256[] private pendingMultisigTransactions;
 
    constructor() public {}
 
    /**
     * @notice <STEP 1> in the Multisignature process: A Request for the execution 
     * of protected funtionality is submitted
     */
    function createMultisigTransaction(uint8 _minSignCount, uint256 _groupMembersCount) 
            internal 
            returns (uint256) {
        bool isMinSignCount = (_minSignCount < _groupMembersCount);
        Erequire(isMinSignCount, "Minimum signature count must be less than the amount of group members.");
 
        uint256 transactionId = transactionIdx++;
        MultisigTransaction memory msTransaction;
        msTransaction.initiatedBy = msg.sender;
        msTransaction.signatureCount = 0;
        msTransaction.minSignatureCount = _minSignCount;
        multisigTransactions[transactionId] = msTransaction;
        pendingMultisigTransactions.push(transactionId);
 
        emit MultisigExecutionRequest(msg.sender, transactionId);
        
        return transactionId;
    }
 
    /**
     * @notice <STEP 2> in the Multisignature process: Using the transactionId 
     * given in <STEP 1>, the request is signed.
     */
    function signMultisigTransaction(uint256 _transactionId) 
            internal 
            returns (uint8 _signCount, uint8 _minSignCount) { 
        MultisigTransaction storage msTransaction = multisigTransactions[_transactionId];
        Erequire(0x0 != msTransaction.initiatedBy, "Transaction must exist");
        require(msTransaction.signatures[msg.sender] == 0, "Already signed transaction");
 
        msTransaction.signatures[msg.sender] = 1;
        msTransaction.signatureCount++;
 
        emit MultisigExecutionApproval(msg.sender, _transactionId);
 
        _signCount = msTransaction.signatureCount;
        _minSignCount = msTransaction.minSignatureCount;
 
        if (msTransaction.signatureCount >= msTransaction.minSignatureCount) { 
            // <STEP 3> Once the minimum amount of signed transactions are reached, 
            // the protected functionality should be triggered.
            emit MultisigExecutionCompleted(_transactionId);
            deleteMultisigTransaction(_transactionId);
        }
    }
 
    /**
     * @notice <STEP 4> Once the multi-signature process is finished or a cancellation is
     * requested, the transaction is deleted
     */
    function deleteMultisigTransaction(uint256 _transactionId) 
            internal {
        Erequire(pendingMultisigTransactions.length > 0, "There are no transaction to delete");
 
        uint8 replace = 0;
        for (uint256 i = 0; i < pendingMultisigTransactions.length; i++) {
            Iif (1 == replace) {
                pendingMultisigTransactions[i-1] = pendingMultisigTransactions[i];
            } else Eif (pendingMultisigTransactions[i] == _transactionId) {
                replace = 1;
            }
        }
        Eassert(replace == 1);
 
        // Created an Overflow
        delete pendingMultisigTransactions[pendingMultisigTransactions.length-1];
        pendingMultisigTransactions.length--;
        delete multisigTransactions[_transactionId];
    }
 
    /**
     * @notice Cancels the current transaction, at a cost.
     */
    function cancelMultisigTransaction(uint256 _transactionId) 
            internal {
        emit MultisigExecutionCancelled(_transactionId);
        deleteMultisigTransaction(_transactionId);
    }
 
    /**
     * @notice Returns the number of pending transactions.
     */
    function getPendingConfirmations() 
            public   
            view 
            returns (uint256) { 
        return pendingMultisigTransactions.length;
    }
}
