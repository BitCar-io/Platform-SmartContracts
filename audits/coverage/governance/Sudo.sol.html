pragma solidity ^0.4.24;
 
import "../libs/rbac/RBACData.sol";
import "./AccessControl.sol";
import "./Multisignature.sol";
 
/**
 * @title Sudo
 * @notice The Sudo contract has a SUDO Group of addresses, 
 * and provides basic authorization control functions.
 * This simplifies the implementation of "user permissions".
 */
contract Sudo is AccessControl, Multisignature {
 
    address private tempAddr;
 
    /**
     * @notice Constructor needs instance of RBACData 
     * @param _config Reference to the RBACData contract
     */
    constructor(address _config)
            public 
            AccessControl(AccessControl.SUDO_ROLE, _config) {
    }
 
    /**
     * @notice Adds address to Sudo Group
     * @param _addr address
     */
    function addAddressToSudoGroup(address _addr)
            public 
            onlySudoOrOwner {
        addAddressToGroup(_addr);
        verifyUser(_addr);
    }
 
    /**
     * @notice Requests execution of address removal from Sudo Group
     * @param _addr address
     */
    function requestRemoveAddressFromSudoGroup(address _addr)
            public 
            onlySudo 
            returns (uint256) {
        Erequire(_addr != address(0), "Address needs to be a valid one.");
        tempAddr = _addr;
        return createMultisigTransaction(2, getRoleMembersCount());
    }
 
    /**
     * @notice Once confirmed, the address is removed.
     * @param _transactionId The transaction ID obtained from the execution of the 
     * function requestRemoveAddressFromSudoGroup
     */
    function confirmRemoveAddressFromSudoGroup(uint256 _transactionId)
            public 
            onlySudo {
        uint8 signCount;
        uint8 minSignCount;
        (signCount, minSignCount) = signMultisigTransaction(_transactionId);
        if (signCount >= minSignCount) {
            removeAddressFromGroup(tempAddr);
        }
    }
}
