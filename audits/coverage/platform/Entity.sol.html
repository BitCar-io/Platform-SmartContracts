/*
* Entity
* Manages a platform entity such as Users or Assets
* Notes:
*/
 
pragma solidity ^0.4.24;
 
import "../governance/Groups.sol";
 
contract Entity is Groups {
 
    address owner;
 
    modifier hasFeedback() {
        require(hasPositiveFeedback(), "Entity does not have a positive feedback");
        _;
    }
 
    uint256 likes;
    uint256 dislikes;
 
    // CONSTRUCTOR
    constructor(address _config) 
            public 
            Groups(_config) {
        Erequire(_config != address(0));
        
        owner = msg.sender;
    }
 
    /////////////
    // SETTERS
    /////////////
 
    // TODO: Prevent from user liking itself
    function like() public {
        likes++;
    }
 
    function dislike() public {
        dislikes++;
    }
 
    /////////////
    // GETTERS
    /////////////
 
    function getLikes() public view returns(uint256) {
        return likes;
    }
 
    function getDislikes() public view returns(uint256) {
        return dislikes;
    }
 
    function hasPositiveFeedback() public view returns(bool) {
        if (getLikes() > getDislikes()) {
            return true;
        }
 
        return false;
    }
}
