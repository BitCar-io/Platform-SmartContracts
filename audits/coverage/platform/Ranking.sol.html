pragma solidity ^0.4.24;
 
import "../platform/Registry.sol";
 
import "../factories/AssetRankFactory.sol";
import "../libs/interfaces/IAssetRank.sol";
 
contract Ranking {
    IAssetRank private assetRank;
 
    Registry config;
 
    /**
     * @notice Constructor
     * @param _config The Registry address
     */
    constructor(address _config) public {
 
        Erequire(_config != address(0));
 
        config = Registry(_config);
    }
 
    /**
     * @notice If the AssetRank contract instance hasn´t been created, create it
     */
    function createAssetRankInstance() private {
        if (address(assetRank) == address(0)) {
            AssetRankFactory assetRankFactory = AssetRankFactory(config.getAddress("AssetRankFactory"));
            assetRank = assetRankFactory.create();
        }
    }
 
    /**
     * @notice Updates all values for the ranking process
     * @param _newInput Array with all the input values for the Ranking algorithm
     */
    function setRankInputs(uint256[] _newInput) public {
        createAssetRankInstance();
        assetRank.setRankInputs(_newInput);
    }
 
    /**
     * @notice Calculations behind the ranking process of the asset in question.
     * @return _ranking The score of the ranking process
     *         _aboveThreshold TRUE if the score is >= The Threshold, FALSE otherwise
     *         _threshold The Threshold
     *         _unit The Unit used for all the calculations
     */
    function calculateRanking() 
            public 
            returns (uint256 _ranking, bool _aboveThreshold, uint256 _threshold, uint256 _unit) {
        createAssetRankInstance();
        uint256 ranking;
        bool aboveThreshold;
        uint256 threshold;
        uint256 unit;
        (ranking, aboveThreshold, threshold, unit) = assetRank.calculateRanking();
        _ranking = ranking;
        _aboveThreshold = aboveThreshold;
        _threshold = threshold;
        _unit = unit;
    }
}
