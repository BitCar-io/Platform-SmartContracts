pragma solidity ^0.4.24;
 
import "./Portfolio.sol";
 
/**
 * @title GenericPortfolio
 */
contract GenericPortfolio is Portfolio {
 
    /**
     * @param _config Registry instance
     */
    constructor(address _config) 
            public 
            Portfolio(Portfolio.GENERIC, _config) {
    }
 
    /**
     * @dev Adds a new Token to the map
     * @param _wallet The Wallet to get the Token from
     * @param _addr The Address being stored
     */
    function addAddress(address _wallet, address _addr) 
            public 
            onlyWhitelisted {
        addItem(_wallet, _addr);
    }
 
    /**
     * @dev Gets an Address from the map
     * @param _index The index where the Address is stored
     */
    function getAddress(address _wallet, uint256 _index) 
            public 
            view 
            returns(address) {
        return getItem(_wallet, _index);
    }
 
    /**
     * @dev TRUE if the address belongs to the _wallet, FALSE otherwise.
     * @param _wallet The source _wallet 
     * @param _addr The address being tested
     */
    function hasAddress(address _wallet, address _addr) 
            public 
            view 
            returns(bool) {
        return idxedEntities.hasItem(_wallet, addressType, _addr); 
    }
}
