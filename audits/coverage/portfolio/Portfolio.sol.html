pragma solidity ^0.4.24;
 
import "../libs/ownership/Ownable.sol";
import "../lifecycle/Pausable.sol";
import "../libs/IndexedEntities.sol";
 
/**
 * @title Portfolio
 */
contract Portfolio is Ownable, Pausable {
 
    string public constant TOKEN = "TOKEN";
    string public constant CAR = "CAR";
    string public constant GENERIC = "GENERIC";
    IndexedEntities internal idxedEntities;
    string internal addressType;
 
    /**
     * @param _config Registry instance
     */
    constructor(string _addressType, address _config) 
            public 
            Pausable(_config) {
        idxedEntities = IndexedEntities(Registry(_config).getAddress("IndexedEntities"));
        addressType = _addressType;
    }
 
    /**
     * @dev Adds a new Token to the map
     * @param _wallet The Wallet to get the Token from
     * @param _addr The Token being stored
     */
    function addItem(address _wallet, address _addr) 
            internal 
            whenNotPaused {
        idxedEntities.addItem(_wallet, addressType, _addr);
    }
 
    /**
     * @dev Gets a Token from the map
     * @param _index The index where the Token is stored
     */
    function getItem(address _wallet, uint256 _index) 
            internal 
            view 
            returns(address) {
        return idxedEntities.getItem(_wallet, addressType, _index);
    }
 
    /**
     * @dev Returns the quantity of addresses related to the _wallet.
     * @param _wallet The wallet being queried 
     */
    function getCount(address _wallet) 
            public 
            view 
            returns(uint256) {
        return idxedEntities.getCount(_wallet, addressType);
    }
 
    /**
     * @dev Returns the type of addresses stored
     */
    function getType() 
            public 
            view 
            returns(string) {
        return addressType;
    }
}
