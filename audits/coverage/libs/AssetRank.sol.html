/**
 * Because in solidity there are no floating point types, integers must be used.
 * So, for the purpose of using "normalized" values for all variables, we use
 * 1000000 ( 1 million / 1,000,000 ) as the unit (1):
 * 1 : 1,000,000;  0.5 : 500,000;  0.01 : 10,000;  0.001 : 1,000;  2.5 : 2,500,000 , so on and so forth
 */
 
pragma solidity ^0.4.24;
 
import "./interfaces/IAssetRank.sol";
import "./AssetRankParameters.sol";
import "../vendor/SafeMath.sol";
import "../governance/Groups.sol";
 
/**
 * @title AssetRank
 */
contract AssetRank is IAssetRank, Groups {
 
    using SafeMath for uint256;
 
    AssetRankParameters internal params;
    uint256[] internal input;
 
    /**
     * @notice Event triggered with the results of the ranking process
     */
    event RankResult(uint256 _ranking, bool _aboveThreshold, uint256 _threshold, uint256 _unit);
 
    /**
     * @param _config The Registry address
     */
    constructor(address _config) 
            public 
            Groups(_config) {
        params = AssetRankParameters(Registry(_config).getAddress("AssetRankParameters"));
        for (uint256 p = 0; p < params.getNumParams(); p++) {input.push(0);}
    }
 
    /**
     * @notice Updates all values for the ranking process
     * @param _newInput Array with all the input values for the Ranking algorithm
     */
    function setRankInputs(uint256[] _newInput) 
            external {
        Erequire(_newInput.length == params.getNumParams(), "Qty of parameters out of range");
        for (uint256 p = 0; p < _newInput.length; p++) {input[p] = _newInput[p];}
    }
 
    /**
     * @notice Calculations behind the ranking process of the asset in question.
     * @return _ranking The score of the ranking process
     *         _aboveThreshold TRUE if the score is >= The Threshold, FALSE otherwise
     *         _threshold The Threshold
     *         _unit The Unit used for all the calculations
     */
    function calculateRanking() 
            external 
            returns (uint256 _ranking, bool _aboveThreshold, uint256 _threshold, uint256 _unit) {
        // The current model uses 25 different variables to calculate the ranking of a car
        _unit = params.getTheUnit();
        _threshold = params.getTheThreshold();
        uint256 product = _unit;
        uint256 sum = 0;
        for (uint8 l = 0; l < 12; l++) {product = (product.mul((input[l].mul(params.getParam(l))).div(_unit))).div(_unit);}
        for (uint8 m = 12; m < params.getNumParams(); m++) {sum = sum.add(((input[m].mul(params.getParam(m))).div(_unit)));}
        _ranking = product.mul(sum).div(_unit);
        _aboveThreshold = (_ranking >= _threshold) ? true : false;
        emit RankResult(_ranking, _aboveThreshold, _threshold, _unit);
    }
}
